1. `pnpm dev` locally: home page loads, magic link can be sent and caught in Mailpit, clicking link logs you in, `/dashboard` shows your email.
2. `pnpm test` and `pnpm test:e2e` both green.
3. GitHub Actions CI green on first push.
4. Railway staging URL (e.g. `v2.datapatch.net` or Railway default domain) serves the app; magic link sent via Resend arrives in a real inbox; login succeeds.

- [ ] **Step 6.2: Create `src/lib/env.ts`**

```ts
import { z } from 'zod';

const envSchema = z.object({
  NODE_ENV: z.enum(['development', 'test', 'production']).default('development'),
  DATABASE_URL: z.string().url(),
  NEXTAUTH_SECRET: z.string().min(32),
  NEXTAUTH_URL: z.string().url(),
  EMAIL_SERVER_HOST: z.string().min(1),
  EMAIL_SERVER_PORT: z.coerce.number().int().positive(),
  EMAIL_SERVER_USER: z.string().optional().default(''),
  EMAIL_SERVER_PASSWORD: z.string().optional().default(''),
  EMAIL_FROM: z.string().email(),
  REDIS_URL: z.string().url(),
});

export type Env = z.infer<typeof envSchema>;

function loadEnv(): Env {
  const parsed = envSchema.safeParse(process.env);
  if (!parsed.success) {
    const formatted = parsed.error.flatten().fieldErrors;
    console.error('❌ Invalid environment variables:', formatted);
    throw new Error('Environment validation failed — see logs above.');
  }
  return parsed.data;
}

export const env: Env = loadEnv();
```

import { describe, it, expect, beforeEach } from 'vitest';
import { z } from 'zod';

// We re-define here to test schema behavior in isolation, because
// src/lib/env.ts reads process.env at import time.
const testSchema = z.object({
  DATABASE_URL: z.string().url(),
  NEXTAUTH_SECRET: z.string().min(32),
});

describe('env schema', () => {
  it('rejects a secret shorter than 32 chars', () => {
    const result = testSchema.safeParse({
      DATABASE_URL: 'postgresql://x:y@localhost:5432/db',
      NEXTAUTH_SECRET: 'too-short',
    });
    expect(result.success).toBe(false);
  });

  it('rejects a non-URL DATABASE_URL', () => {
    const result = testSchema.safeParse({
      DATABASE_URL: 'not-a-url',

    expect(result.success).toBe(false);

  it('accepts a valid config', () => {
    const result = testSchema.safeParse({
      DATABASE_URL: 'postgresql://x:y@localhost:5432/db',
      NEXTAUTH_SECRET: 'a'.repeat(32),

const MAILPIT_API = 'http://localhost:8025/api/v1';

test.beforeEach(async () => {
  const ctx = await request.newContext();
  await ctx.delete(`${MAILPIT_API}/messages`);
  await ctx.dispose();

test('magic link flow: request → email → click → dashboard', async ({ page, request: req }) => {
  const email = `e2e-${Date.now()}@example.com`;

  await page.goto('/signin');
  await page.getByPlaceholder('you@example.com').fill(email);
  await page.getByRole('button', { name: /send magic link/i }).click();

  await expect(page.getByText(/check your email/i)).toBeVisible();

  // Poll Mailpit for the email (up to 10s).
  let link: string | undefined;
  for (let i = 0; i < 20; i++) {
    const res = await req.get(`${MAILPIT_API}/messages`);
    const body: { messages: Array<{ ID: string; To: Array<{ Address: string }> }> } = await res.json();
    const msg = body.messages.find((m) => m.To.some((t) => t.Address === email));
    if (msg) {
      const detail = await req.get(`${MAILPIT_API}/message/${msg.ID}`);
      const html: string = (await detail.json()).HTML;
      const match = html.match(/href="([^"]+\/api\/auth\/callback\/nodemailer[^"]+)"/);
      if (match) {
        link = match[1].replace(/&amp;/g, '&');
        break;
      }
    }
    await new Promise((r) => setTimeout(r, 500));
  }

  expect(link, 'magic link not found in Mailpit').toBeDefined();

  await page.goto(link!);
  await expect(page).toHaveURL(/\/dashboard$/);
  await expect(page.getByText(email)).toBeVisible();
});

```json

  "build": {
    "builder": "DOCKERFILE",

    "startCommand": "npx prisma migrate deploy && node server.js",
    "healthcheckPath": "/api/health",

    "restartPolicyMaxRetries": 3

}
```
